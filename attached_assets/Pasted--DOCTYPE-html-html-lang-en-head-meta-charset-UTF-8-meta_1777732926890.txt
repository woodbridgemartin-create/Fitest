<script>
const questions = [
  "I feel mentally fresh at the start of most days",
  "I sleep well consistently",
  "My energy is stable throughout the day",
  "I exercise regularly",
  "My stress is manageable",
];

let answers = [];

function startAudit() {
  document.getElementById("audit").classList.remove("hidden");
  window.scrollTo(0, document.body.scrollHeight);
  renderQuestions();
}

function renderQuestions() {
  const container = document.getElementById("questions");
  container.innerHTML = "";

  questions.forEach((q, i) => {
    const div = document.createElement("div");
    div.className = "mb-6";

    div.innerHTML = `
      <p class="mb-2">${q}</p>
      <input type="range" min="1" max="5" value="3" 
        onchange="answers[${i}] = this.value"
        class="w-full">
    `;

    container.appendChild(div);
  });
}

function submitAudit() {
  let total = answers.reduce((a, b) => a + Number(b || 3), 0);
  let score = Math.round((total / (questions.length * 5)) * 100);

  let tier = "";
  let color = "";

  if (score <= 25) {
    tier = "Critical";
    color = "text-red-400";
  } else if (score <= 50) {
    tier = "Exposed";
    color = "text-yellow-400";
  } else if (score <= 75) {
    tier = "Performing";
    color = "text-lime-400";
  } else {
    tier = "Elite";
    color = "text-green-400";
  }

  document.getElementById("score").innerText = score + "/100";
  document.getElementById("tier").innerText = tier;
  document.getElementById("tier").className = color + " text-2xl font-bold mt-4";

  document.getElementById("audit").classList.add("hidden");
  document.getElementById("result").classList.remove("hidden");
}
</script>
